def calculate_statistics(data_list):
    # 비효율적인 리스트 연산
    result = []
    for i in range(len(data_list)):
        # 반복적인 계산
        mean = sum(data_list) / len(data_list)
        # 불필요한 리스트 생성
        filtered = [x for x in data_list if x > mean]
        result.append(sum(filtered))
    return result
